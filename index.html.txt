  var step1Fields=['f-nombre','f-email','f-telefono','f-edad'];
  function isFieldOk(id){var el=document.getElementById(id);if(!el||!el.value.trim())return false;if(id==='f-email')return/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(el.value.trim());if(id==='f-edad'){var a=parseInt(el.value);return!isNaN(a)&&a>=18&&a<=99;}return true;}
  function checkStep1(){var ok=step1Fields.every(isFieldOk);var btn=document.getElementById('btn-next');if(btn){btn.disabled=!ok;btn.style.opacity=ok?'1':'0.4';}return ok;}
  step1Fields.forEach(function(id){var el=document.getElementById(id);if(el){el.addEventListener('input',checkStep1);el.addEventListener('blur',checkStep1);}});

  var btnNext=document.getElementById('btn-next');
  var btnBack=document.getElementById('btn-back');
  var step1El=document.getElementById('form-step-1');
  var step2El=document.getElementById('form-step-2');

  if(btnNext){btnNext.addEventListener('click',function(){if(!checkStep1())return;step1El.style.opacity='0';setTimeout(function(){step1El.style.display='none';step2El.style.display='block';setTimeout(function(){step2El.style.opacity='1';},30);document.getElementById('booking').scrollIntoView({behavior:'smooth',block:'start'});},300);});}
  if(btnBack){btnBack.addEventListener('click',function(){step2El.style.opacity='0';setTimeout(function(){step2El.style.display='none';step1El.style.display='block';setTimeout(function(){step1El.style.opacity='1';},30);document.getElementById('booking').scrollIntoView({behavior:'smooth',block:'start'});},250);});}

  // Submit button enable
  var legalCheck=document.getElementById('legal-check');
  var textarea=document.getElementById('idea-textarea');
  function checkStep2(){
    var tamOk=document.getElementById('f-tamano')&&document.getElementById('f-tamano').value.trim().length>0;
    var ideaOk=textarea&&textarea.value.length>=20;
    var checkOk=legalCheck&&legalCheck.checked;
    var btn=document.getElementById('submit-btn');
    if(btn){btn.disabled=!(tamOk&&ideaOk&&checkOk);btn.style.backgroundColor=(tamOk&&ideaOk&&checkOk)?'#E11D2E':'#222';}
  }
  if(document.getElementById('f-tamano'))document.getElementById('f-tamano').addEventListener('input',checkStep2);
  if(textarea)textarea.addEventListener('input',function(){checkStep2();this.style.height='auto';this.style.height=this.scrollHeight+'px';});
  if(legalCheck)legalCheck.addEventListener('change',checkStep2);

  // Body zones interactive
  document.querySelectorAll('.bzone').forEach(function(z){z.addEventListener('click',function(){this.classList.toggle('selected');});});

  // Floating CTA
  var floatingBtn=document.createElement('button');
  floatingBtn.setAttribute('onclick',"document.getElementById('booking').scrollIntoView({behavior:'smooth'})");
  floatingBtn.style.cssText='display:none;position:fixed;left:24px;right:24px;bottom:24px;z-index:4000;padding:18px 32px;border-radius:14px;background:#E11D2E;border:none;cursor:pointer;box-shadow:0 4px 24px rgba(225,29,46,0.35);';
  floatingBtn.innerHTML='<span style="font-size:11px;font-weight:700;text-transform:uppercase;letter-spacing:0.25em;color:white;">TERMINE</span>';
  document.body.appendChild(floatingBtn);
  function updateFloat(){var bk=document.getElementById('booking');var bkr=bk?bk.getBoundingClientRect():null;var bkv=bkr&&bkr.top<window.innerHeight&&bkr.bottom>0;if(bkv||(window.innerHeight+window.scrollY)>=document.documentElement.scrollHeight-80){floatingBtn.style.display='none';}else{floatingBtn.style.display='block';}}
  updateFloat();
  window.addEventListener('scroll',updateFloat,{passive:true});
});

function closeModal(id){document.getElementById('modal-'+id).classList.remove('active');document.body.style.overflow='auto';}
