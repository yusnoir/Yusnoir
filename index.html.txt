// ── AUTOSAVE ────────────────────────────────────────────────────────────
(function() {
  var STORAGE_KEY = 'yusnoir_form';
  var fields = [
    { id: 'f-nombre',   key: 'nombre' },
    { id: 'f-email',    key: 'email' },
    { id: 'f-telefono', key: 'telefono' },
    { id: 'f-edad',     key: 'edad' },
    { id: 'f-tamano',   key: 'tamano' },
    { id: 'f-idea',     key: 'idea' }
  ];

  function save() {
    var data = {};
    fields.forEach(function(f) {
      var el = document.getElementById(f.id);
      if (el) data[f.key] = el.value;
    });
    var zona = document.getElementById('zona-input');
    if (zona) data.zona = zona.value;
    try { localStorage.setItem(STORAGE_KEY, JSON.stringify(data)); } catch(e) {}
  }

  function restore() {
    var raw;
    try { raw = localStorage.getItem(STORAGE_KEY); } catch(e) {}
    if (!raw) return;
    var data;
    try { data = JSON.parse(raw); } catch(e) { return; }
    var anyRestored = false;
    fields.forEach(function(f) {
      var el = document.getElementById(f.id);
      if (el && data[f.key]) {
        el.value = data[f.key];
        el.dispatchEvent(new Event('input'));
        anyRestored = true;
      }
    });
    if (data.zona && data.zona.trim()) {
      var zones = data.zona.split(', ');
      zones.forEach(function(z) {
        if (z) setTimeout(function(){ if(window.toggleZone) window.toggleZone(z); }, 200);
      });
      anyRestored = true;
    }
    if (anyRestored) {
      var banner = document.getElementById('autosave-banner');
      if (banner) {
        banner.style.display = 'flex';
        setTimeout(function() {
          banner.style.opacity = '0';
          setTimeout(function() { banner.style.display='none'; banner.style.opacity='1'; }, 600);
        }, 3500);
      }
    }
  }

  function wireAutosave() {
    fields.forEach(function(f) {
      var el = document.getElementById(f.id);
      if (el) {
        el.addEventListener('input', save);
        el.addEventListener('change', save);
      }
    });
    var zona = document.getElementById('zona-input');
    if (zona) { zona.addEventListener('change', save); zona.addEventListener('change', function(){ if(typeof checkStep2==='function') checkStep2(); }); }
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', function() { wireAutosave(); restore(); });
  } else {
    wireAutosave();
    restore();
  }
})();

// ── FIELD CHAIN (Enter/Next) ─────────────────────────────────────────────
(function() {
  var chain = ['f-nombre', 'f-email', 'f-telefono', 'f-edad', 'f-tamano', 'idea-textarea'];
  document.addEventListener('keydown', function(e) {
    if (e.key !== 'Enter') return;
    var active = document.activeElement;
    if (!active || !active.id) return;
    var idx = chain.indexOf(active.id);
    if (idx === -1) return;
    if (active.tagName === 'TEXTAREA') return;
    e.preventDefault();
    var nextId = chain[idx + 1];
    if (nextId) {
      var next = document.getElementById(nextId);
      if (next) next.focus();
    } else {
      active.blur();
    }
  });
})();

          ev.stopPropagation();
          var i = selectedZones.indexOf(this.getAttribute('data-remove'));
          if (i !== -1) selectedZones.splice(i, 1);
          hideLimitMsg();
          renderSelection();
        });
        tag.appendChild(document.createTextNode(z));
        tag.appendChild(btn);
        txt.appendChild(tag);
      });
    } else {
      box.style.display = 'none';
    }
    var label = document.getElementById('bodymap-label');
    var base = bmlabels[currentLangBM] || bmlabels.de;
    label.textContent = selectedZones.length > 0 ? base + ' (' + selectedZones.length + '/' + MAX + ')' : base;
    if (typeof checkStep2 === 'function') checkStep2();
  }

  function showLimitMsg() {
    var el = document.getElementById('zone-limit-msg');
    if (!el) return;
    el.textContent = limitMsgs[currentLangBM] || limitMsgs.de;
    el.style.display = 'block';
    clearTimeout(el._t);
    el._t = setTimeout(function() { el.style.display = 'none'; }, 2500);
  }
  function hideLimitMsg() {
    var el = document.getElementById('zone-limit-msg');
    if (el) el.style.display = 'none';
  }

  document.addEventListener('langchange', function(e) {
    currentLangBM = e.detail;
    var label = document.getElementById('bodymap-label');
    if (label) {
      var base = bmlabels[e.detail] || bmlabels.de;
      label.textContent = selectedZones.length > 0 ? base + ' (' + selectedZones.length + '/' + MAX + ')' : base;
    }
  });

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', initBodyMap);
  } else {
    initBodyMap();
  }
})();

function toggleFaq(btn) {
  var answer = btn.nextElementSibling;
  var isOpen = answer.classList.contains('open');
  document.querySelectorAll('.faq-a').forEach(function(a){ a.classList.remove('open'); });
  document.querySelectorAll('.faq-q').forEach(function(b){ b.classList.remove('open'); });
  if (!isOpen) { answer.classList.add('open'); btn.classList.add('open'); }
}

var SUCCESS_VIDEO_URL = 'https://www.instagram.com/reel/C9wXWQ9NaKe/';
function loadSuccessVideo(wrap) {
  if (!SUCCESS_VIDEO_URL) return;
  // Si es YouTube: extraer ID y usar iframe embed
  var ytMatch = SUCCESS_VIDEO_URL.match(/(?:youtu\.be\/|v=)([\w-]{11})/);
  if (ytMatch) {
    wrap.innerHTML = '<iframe src="https://www.youtube.com/embed/' + ytMatch[1] + '?autoplay=1&mute=1" style="width:100%;height:100%;border:none;" allowfullscreen allow="autoplay"></iframe>';
    return;
  }
  // Si es Instagram/otro: abrir en nueva pestaña
  window.open(SUCCESS_VIDEO_URL, '_blank');
}
